# Vue еҲқе§ӢеҢ–жёІжҹ“жәҗз Ғи§Јжһҗ

TIP

жӯӨзҜҮдё»иҰҒи®ІдәҶе°Ҷhtmlеӯ—з¬ҰдёІжңҖз»ҲиҪ¬еҢ–жҲҗrenderеҮҪж•°д№ӢеҗҺпјҢеҸҲжҖҺд№Ҳз”ҹжҲҗиҷҡжӢҹdom(vnode)пјҢеҶҚж №жҚ®иҷҡжӢҹdomз”ҹжҲҗзңҹе®һDOMз»“жһ„жёІжҹ“еҲ°йЎөйқўдёҠзҡ„иҝҮзЁӢ

// src/init.js

import { initState } from "./state";
import { mountComponent } from "./lifecycle";
import { compileToFunction } from "./compiler/index";

export function initMixin(Vue) {
  Vue.prototype._init = function(options) {
    // el,data
    const vm = this;
    vm.$options = options; // еҗҺйқўдјҡеҜ№optionsиҝӣиЎҢжү©еұ•ж“ҚдҪң

    // еҜ№ж•°жҚ®иҝӣиЎҢеҲқе§ӢеҢ– watch computed props data ...
    initState(vm); // vm.$options.data  ж•°жҚ®еҠ«жҢҒ

    // еҰӮжһңжңүelе…ғзҙ пјҢе°Ҷж•°жҚ®жёІжҹ“еҲ°жЁЎжқҝдёҠ
    if (vm.$options.el) {
      vm.$mount(vm.$options.el);
    }
  };

  Vue.prototype.$mount = function(el) {
    const vm = this;
    const options = vm.$options;
    el = document.querySelector(el);
    // жҢӮиҪҪж—¶йңҖиҰҒжҠҠдј е…Ҙзҡ„elе…ғзҙ жҢӮиҪҪеҲ°vmзҡ„$elеұһжҖ§дёҠпјҢдёәдәҶд№ӢеҗҺDOMжӣҙж–°жӣҝжҚўDOM
    vm.$el = el;
    // жҠҠжЁЎжқҝиҪ¬еҢ–жҲҗ еҜ№еә”зҡ„жёІжҹ“еҮҪж•°(render) =гҖӢ иҷҡжӢҹdomжҰӮеҝө vnode =гҖӢ diffз®—жі• жӣҙж–°иҷҡжӢҹdom =гҖӢ дә§з”ҹзңҹе®һиҠӮзӮ№пјҢжӣҙж–°
    // еҰӮжһңжңүrender е°ұз”Ёrender
    // жІЎжңүrender зңӢжңүжІЎжңүtemplate  жңүе°ұз”Ё
    // жІЎжңүtemplate е°ұжүҫel
    if (!options.render) {
      // жІЎжңүrenderз”ЁtemplateпјҢзӣ®еүҚжІЎrender
      let template = options.template;
      if (!template && el) {
        // з”ЁжҲ·д№ҹжІЎжңүдј йҖ’template е°ұеҸ–elзҡ„еҶ…е®№дҪңдёәжЁЎжқҝ
        template = el.outerHTML;
      }
      // жңҖеҗҺйңҖиҰҒжҠҠtemplateиҪ¬жҚўжҲҗrenderеҮҪж•°
      let render = compileToFunction(template);
      // з”ҹжҲҗrenderеҮҪж•°еҗҺжҢӮиҪҪеҲ°vmзҡ„optionsеұһжҖ§дёҠ
      options.render = render;
    }
    // options.render е°ұжҳҜжёІжҹ“еҮҪж•°
    // и°ғз”Ёrenderж–№жі• жёІжҹ“жҲҗзңҹе®һdom жӣҝжҚўжҺүйЎөйқўзҡ„еҶ…е®№
    mountComponent(vm, el); // з»„д»¶зҡ„жҢӮиҪҪжөҒзЁӢ
  };
}

// src/lifecycle.js

export function mountComponent(vm, el) {
  // з”ҹжҲҗrenderеҮҪж•°д№ӢеҗҺдјҡи°ғз”ЁжӯӨж–№жі•пјҢvm.optionsдёҠдјҡжңүrenderж–№жі•
  // д№ӢеҗҺи°ғз”Ё vm._render()пјҢдјҡи°ғз”Ёrenderж–№жі• з”ҹжҲҗиҷҡжӢҹdom
  // жңҖеҗҺи°ғз”Ёvm._update() ж №жҚ®иҷҡжӢҹdomз”ҹжҲҗзңҹе®һDOMжёІжҹ“еҲ°йЎөйқўдёҠ
  //   _updateе’Ң._renderж–№жі•йғҪжҳҜжҢӮиҪҪеңЁVueеҺҹеһӢзҡ„ж–№жі•  зұ»дјј_init
  vm._update(vm._render());
}

# _render з”ҹжҲҗиҷҡжӢҹ DOM

// src/render.js
import { createElement, createTextElement } from "./vdom/index";

export function renderMixin(Vue) {
  // з”ҹжҲҗиҷҡжӢҹDOMзҡ„ж–№жі•
  Vue.prototype._render = function() {
    const vm = this;
    // жҲ‘д»¬и§ЈжһҗеҮәжқҘзҡ„renderж–№жі•пјҢд№ҹжңүеҸҜиғҪжҳҜз”ЁжҲ·еҶҷзҡ„
    let render = vm.$options.render;
    // и°ғз”Ёrenderж–№жі•з”ҹжҲҗиҷҡжӢҹDOM
    let vnode = render.call(vm);
    return vnode;
  };

  Vue.prototype._c = function() {
    // з”ҹжҲҗе…ғзҙ зҡ„иҷҡжӢҹиҠӮзӮ№
    return createElement(this, ...arguments);
  };

  Vue.prototype._v = function(text) {
    // з”ҹжҲҗж–Үжң¬зҡ„иҷҡжӢҹиҠӮзӮ№
    return createTextElement(this, text);
  };

  // д»Һvmе®һдҫӢдёҠеҸ–еҖјзҡ„ж–№жі•
  Vue.prototype._s = function(val) {
    // еҰӮжһңvalзҡ„жҳҜдёҖдёӘеҜ№иұЎ  йңҖиҰҒJSON.stringify
    if (typeof val == "object") return JSON.stringify(val);
    return val;
  };
}

// lifecycleж–Үд»¶дёӯеўһеҠ  _updateж–№жі•
import { patch } from "./vdom/patch";

export function lifecycleMixin(Vue) {
  // е°ҶиҷҡжӢҹиҠӮзӮ№иҪ¬жҚўжҲҗзңҹе®һзҡ„dom
  Vue.prototype._update = function (vnode) {

// src/vdom/patch.js

// patchз”ЁжқҘжёІжҹ“е’Ңжӣҙж–°и§Ҷеӣҫ д»ҠеӨ©еҸӘд»Ӣз»ҚеҲқж¬ЎжёІжҹ“зҡ„йҖ»иҫ‘
export function patch(oldVnode, vnode) {
  // ж №жҚ®дј е…Ҙзҡ„oldVnodeзҡ„nodeTypeжҳҜеҗҰжңүеҖјеҲӨж–ӯжҳҜйҰ–ж¬ЎжёІжҹ“иҝҳжҳҜи§Ҷеӣҫжӣҙж–°
  // йҰ–ж¬ЎжёІжҹ“ж—¶пјҢдј е…Ҙзҡ„vm.$elе°ұжҳҜз”ЁжҲ·дј е…Ҙoptionsдёӯзҡ„elйҖүйЎ№  жүҖд»ҘжҳҜзңҹе®һdom
  // и§Ҷеӣҫжӣҙж–°зҡ„ж—¶еҖҷ  vm.$el е°ұиў«жӣҝжҚўжҲҗдәҶжӣҙж–°д№ӢеүҚзҡ„иҖҒзҡ„иҷҡжӢҹdom
  if (oldVnode.nodeType == 1) { // йҰ–ж¬ЎжёІжҹ“зҡ„йҖ»иҫ‘
    const oldElm = oldVnode; // <div id="app">{{name}}<div>
    const parentElm = oldElm.parentNode; // body
    // е°ҶиҷҡжӢҹdomиҪ¬еҢ–жҲҗзңҹе®һdomиҠӮзӮ№
    let el = createElm(vnode);
    // жҸ’е…ҘеҲ° иҖҒзҡ„elиҠӮзӮ№дёӢдёҖдёӘиҠӮзӮ№зҡ„еүҚйқў е°ұзӣёеҪ“дәҺжҸ’е…ҘеҲ°иҖҒзҡ„elиҠӮзӮ№зҡ„еҗҺйқў
    // иҝҷйҮҢдёҚзӣҙжҺҘдҪҝз”ЁзҲ¶е…ғзҙ appendChildжҳҜдёәдәҶдёҚз ҙеқҸжӣҝжҚўзҡ„дҪҚзҪ®
    parentElm.insertBefore(el, oldElm.nextSibling);
    // еҲ йҷӨиҖҒзҡ„elиҠӮзӮ№
    parentElm.removeChild(oldVnode);

    return el;
  }else{
    // diff ж–°ж—§ иҷҡжӢҹиҠӮзӮ№
  }
}

// ж №жҚ®иҷҡжӢҹиҠӮзӮ№еҲӣе»әзңҹе®һзҡ„иҠӮзӮ№
function createElm(vnode) {
  let { tag, data, key, children, text } = vnode;
  //   еҲӨж–ӯиҷҡжӢҹdom жҳҜе…ғзҙ иҠӮзӮ№иҝҳжҳҜж–Үжң¬иҠӮзӮ№
  if (typeof tag === "string") { // е…ғзҙ иҠӮзӮ№зҡ„tagжңүеҖјпјҢж–Үжң¬иҠӮзӮ№жІЎжңүtagеұһжҖ§
    //  иҷҡжӢҹиҠӮзӮ№зҡ„elеұһжҖ§жҢҮеҗ‘зңҹе®һdom
    vnode.el = document.createElement(tag);
    // и§ЈжһҗиҷҡжӢҹdomеұһжҖ§
    updateProperties(vnode);
    // еҫӘзҺҜеӯҗиҠӮзӮ№йҖ’еҪ’еҲӣе»әе№¶е°ҶеӯҗиҠӮзӮ№ж”ҫе…ҘзҲ¶иҠӮзӮ№
    children.forEach((child) => {
      return vnode.el.appendChild(createElm(child));
    });
  } else {
    //   ж–Үжң¬иҠӮзӮ№
    vnode.el = document.createTextNode(text);
  }
  return vnode.el;
}

// и§Јжһҗvnodeзҡ„dataеұһжҖ§ жҳ е°„еҲ°зңҹе®һdomдёҠ
function updateProperties(vnode) {
  let newProps = vnode.data || {};
  let el = vnode.el; //зңҹе®һиҠӮзӮ№
  for (let key in newProps) {
    // ж ·ејҸйңҖиҰҒзү№ж®ҠеӨ„зҗҶ
    if (key === "style") {
      for (let styleName in newProps.style) {
        el.style[styleName] = newProps.style[styleName];
      }
    } else if (key === "class") {
      el.className = newProps.class;
    } else {
      // з»ҷиҝҷдёӘе…ғзҙ ж·»еҠ еұһжҖ§ еҖје°ұжҳҜеҜ№еә”зҡ„еҖј
      el.setAttribute(key, newProps[key]);
    }
  }
}
